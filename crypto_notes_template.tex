\subsection{Notation}

\begin{definition}[Negligible Function]
A function $\mu : \mathbb{N} \rightarrow \mathbb{R}$ is \emph{negligible} if for every
polynomial $p(\cdot)$, there exists $N$ such that for all $n > N$,
\[
\mu(n) < \frac{1}{p(n)}.
\]
\end{definition}

\begin{remark}
Negligible functions capture events that occur with vanishing probability
as the security parameter grows.
\end{remark}

\section{Public Key Encryption}
% --------------------------------------------------

\subsection{Syntax}

\begin{definition}[Public Key Encryption Scheme]
A public key encryption scheme is a tuple of probabilistic polynomial-time algorithms
\[
(\Gen, \Enc, \Dec)
\]
such that:
\begin{itemize}
    \item $\Gen(1^n)$ outputs $(pk, sk)$
    \item $\Enc(pk, m)$ outputs a ciphertext $c$
    \item $\Dec(sk, c)$ outputs a message $m$ or $\bot$
\end{itemize}
\end{definition}

% --------------------------------------------------
\subsection{Algorithm Block}
% --------------------------------------------------

\begin{algorithm}[H]
\caption{Key Generation $\Gen(1^n)$}
\begin{algorithmic}[1]
\State Choose random primes $p, q$
\State Compute $N \gets pq$
\State Output $(pk, sk)$
\end{algorithmic}
\end{algorithm}

% --------------------------------------------------
\subsection{Pseudocode Block}
% --------------------------------------------------

\begin{algorithm}[H]
\caption{Encryption $\Enc(pk, m)$}
\begin{algorithmic}[1]
\Require Public key $pk$, message $m$
\Ensure Ciphertext $c$
\State Sample randomness $r$
\State Compute $c \gets f(pk, m, r)$
\State \Return $c$
\end{algorithmic}
\end{algorithm}

% --------------------------------------------------
\subsection{Security Definitions}
% --------------------------------------------------

\begin{theorem}[Correctness]
For all key pairs $(pk, sk)$ generated by $\Gen$ and all messages $m$,
\[
\Dec(sk, \Enc(pk, m)) = m.
\]
\end{theorem}

\begin{proof}
By construction of the encryption and decryption algorithms,
the randomness used during encryption is recoverable during decryption,
ensuring correctness.
\end{proof}

\section{Claims and Proofs}
% --------------------------------------------------

\begin{claim}
If the underlying problem is computationally hard, then the encryption scheme
is IND-CPA secure.
\end{claim}

\begin{proof}
Assume there exists a PPT adversary that breaks IND-CPA security.
We construct a reduction that uses this adversary to solve the underlying
hard problem with non-negligible probability, yielding a contradiction.
\end{proof}
